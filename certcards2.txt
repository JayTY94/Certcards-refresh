limit the number of retries in a retry decorator by maintaining a counter within the decorator and exiting after reaching the maximum number of retries.

Example:
from functools import wraps

def retry(max_retries):
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            attempts = 0
            while attempts < max_retries:
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    attempts += 1
                    print(f"Attempt {attempts} failed: {e}")
            raise Exception("Maximum retries exceeded")
        return wrapper
    return decorator
